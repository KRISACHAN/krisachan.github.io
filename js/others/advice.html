# 函数节流（throttle）

在实际的开发中，常常会有需要监听一些例如onresize、touchmove、scroll的持续执行的事件，但是我们也并不是一定需要连续不断的执行我们监听的逻辑，而且连续不断的执行我们的逻辑，还有可能会引发一系列性能的问题，所以这个时候我们就需要用上函数节流这个功能。函数节流就是预先定义一个函数的执行周期，只有时间间隔等于或大于这个周期的时候才会被执行。

# 函数节流实现

(function () {
    'use strict';
    /*
     * @func
     * @name throttle
     * @desc 函数节流，用来指定函数执行间隔
     * @param {function} fn 要执行的函数逻辑
     * @param {number} interval 函数最小执行间隔
     */
    var throttle = function (fn, interval) {
        var _FN = fn;
        var timer = null;
        var work = true;
        return function () {
            var args = arguments,
               _THIS = this;
            if (work) {
                _FN.apply(_THIS, args);
                return work = false;
            };
            if (timer) {
                return false;
            };
            timer = setTimeout(function () {
                clearTimeout(timer);
                timer = null;
                _FN.apply(_THIS, args);
            }, interval || 500);
        };
    };
    var _global = (function () {
        return this || (0, eval)('this');
    })();
    if (typeof module !== 'undefined' && module.exports) {
        module.exports = throttle;
    } else if (typeof define === 'function' && define.amd) {
        define([], function () {
            return throttle;
        });
    } else {
        !('throttle' in _global) && (_global.throttle = throttle);
    };
})();

# 函数防抖（debounce）

函数防抖（debounce）与函数节流（throttle）作用类似，不过区别在于函数防抖在定义好最小时间间隔之后，如果在这个间隔内有调用，则重新计算执行时间。意思就是在只有在这个时间间隔内，完全不调用这个函数，那么到了时间之后才能继续执行，否则又得重新计算。

# 函数防抖实现

(function () {
    'use strict';
    /*
     * @func
     * @name debounce
     * @desc 函数防抖，用来指定函数执行间隔
     * @param {function} fn 要执行的函数逻辑
     * @param {number} interval 函数最小执行间隔
     */
    var debounce = function (fn, interval) {
        var timer = null;
        return function () {
            clearTimeout(timer);
            timer = setTimeout(() => {
                fn();      
            }, interval);
        };
    };
    var _global = (function () {
        return this || (0, eval)('this');
    })();
    if (typeof module !== 'undefined' && module.exports) {
        module.exports = debounce;
    } else if (typeof define === 'function' && define.amd) {
        define([], function () {
            return debounce;
        });
    } else {
        !('debounce' in _global) && (_global.debounce = debounce)
    };
})();

# 事件委托

首先我们知道DOM事件流有三个阶段：捕获、目标、冒泡。而事件委托则是利用了DOM事件流的特性，通过绑定一个节点来影响子或父的节点。

# 代码如下

<!--
HTML结构
-->
<ul id="list">
    <li class="item-1">1</li>
    <li class="item-2">2</li>
    <li class="item-3">3</li>
    <li class="item-4">4</li>
    <li class="item-5">5</li>
</ul>
<p id="text">文本</p>

<!--
javascript 代码
-->
<script>
    'use strict';
    list.addEventListener('click', event => {
        if (event.target.className.indexOf('item-') === 0) {
            text.innerHTML = event.target.innerHTML;
        };
    });
</script>

# PC端与移动端的事件统一

在以往的开发中，如果我们开发的是响应式的页面，对于一些屏幕交互，我们往往需要写两份代码或者判断是否需要使用mouse或touch事件。但是如今，我们可以简化这道工序了。

# PointerEvent

首先我们来认识一下PointerEvent。PointerEvent 接口代表了由 指针 引发的DOM事件的状态，包括接触点的位置，引发事件的设备类型，接触表面受到的压力等。具体的API解释以及使用规则稍后会另开文章专门进行讲解，有兴趣的可以关注鱼头我的博客动态。

# 例子如下

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <meta name="viewport" content="width=device-width,initial-scale=1.0,maximum-scale=1.0,user-scalable=no" />
        <meta name="screen-orientation" content="portrait">
        <meta name="x5-orientation" content="portrait">
        <meta http-equiv="X-UA-Compatible" content="IE=edge,chrome=1"/>
        <meta http-equiv="Cache-Control" content="no-siteapp">
        <title>路线记忆</title>
        <style>
            * {
                margin: 0;
                padding: 0;
            }
            html,
            body {
              width: 100vw;
              height: 100vh;
              overflow: hidden;
            }
            h1 {
              font-size: 5vw;
              position: absolute;
              text-align: center;
              width: 100%;
            }
            .大圆 {
                width: 20vw;
                height: 20vw;
                border-radius: 50%;
                box-shadow: 0 0 10px rgba(66,132,132,0.6);
                background: #FF8442;
                position: absolute;
                top: 0;
                left: 0;
                cursor: pointer;
                display: none;
            }
            .小圆 {
                width: 10vw;
                height: 10vw;
                border-radius: 50%;
                margin: auto;
                position: absolute;
                left: 50%;
                top: 50%;
                -webkit-transform: translate(-50%, -50%);
                   -moz-transform: translate(-50%, -50%);
                    -ms-transform: translate(-50%, -50%);
                     -o-transform: translate(-50%, -50%);
                        transform: translate(-50%, -50%);
                background: #E9E9E9;
            }
        </style>
    </head>
    <body>
        <h1>请按住屏幕并移动</h1>
        <div class="大圆">
            <div class="小圆"></div>
        </div>
        <script>
            'use strict';
            const 大圆 = document.querySelector('.大圆');
            const 小圆 = document.querySelector('.小圆');
            const 大圆的高 = window.getComputedStyle(大圆, null).height;
            const 大圆的宽 = window.getComputedStyle(大圆, null).width;
            const 大圆的样式 = 大圆.style;
            let 坐标数组 = [];
            let 数组下标 = 0;
            let 定时器 = null;
            let 开关 = true;
            const 运行 = 事件 => {
                console.log(事件);
                let X = 事件.clientX - parseInt(大圆的宽)/2;
                let Y = 事件.clientY - parseInt(大圆的高)/2;
                [].push.call(坐标数组, [X, Y]);
                大圆的样式.left = `${X}px`;
                大圆的样式.top = `${Y}px`;
                大圆的样式.display = `block`;
                数组下标 = 坐标数组.length;
                开关 = true;
            };

            window.addEventListener('pointerdown', 事件 => {
                console.log('pointerdown');
                [].splice.call(坐标数组, 0, 坐标数组.length);
                window.addEventListener('pointermove', 运行);
            });

            window.addEventListener('pointerup', 事件 => {
                console.log('pointerup');
                clearInterval(定时器);
                window.removeEventListener('pointermove', 运行);
                定时器 = setInterval(() => {
                    if (开关) {
                        数组下标--;
                        大圆的样式.left = `${坐标数组[数组下标][0]}px`;
                        大圆的样式.top = `${坐标数组[数组下标][1]}px`;
                        if (数组下标 <= 1) {
                            [].splice.call(坐标数组, 0, 坐标数组.length);
                            大圆的样式.display = `none`;
                            开关 = false;
                        };
                    };
                }, 20);
            });
        </script>
    </body>
</html>
